1. C

#include <stdio.h>
int main() {
    int a;
    scanf("%d", &a);
    
    for(int i=1;i<a+1;i++){
        for(int k=a-i;k>0;k--)
            printf(" ");
        for(int j=1;j<(i*2);j++)
            printf("*");
        printf("\n");
    }
}
